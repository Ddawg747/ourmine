\begin{algorithmic}
\Procedure {RunHyperPipes}{$Nothing$}
\State $HyperPipes := array[]$
\State $Guessed := 0$
\State $GuessedCorrect := 0$
\ForAll {Line in DataFile}
\State $Guessed++$
\State $GuessedCorrect \mbox{+=} Classify(MyHyperPipes,Line)$
\State $HyperPipes := AddExperience(Line,HyperPipes)$
\EndFor
\State $Accuracy := GuessedCorrect/Guessed$
\EndProcedure
\end{algorithmic}

\begin{program}
\begin{algorithmic}
\Procedure {Classify}{$HyperPipes$,$Line$}
\State{$BestScore := 0$}
\State{$BestClass := []$}
\ForAll{HPipe in HyperPipes}
\State{$HPipeScore := 0$}
\ForAll{Attr in Line}
\If{$Attr <= HPipe.Attr.max\; \&\&\; Attr >= HPipe.Attr.min$}
\State{$HPipeScore++$}
\EndIf
\EndFor
\If{$HPipeScore >= BestScore$}
\If{$HPipeScore := BestScore$}
\State{$BestClass := BestClass + HPipe.class$}
\Else
\State{$BestClass := array(HPipe.class)$}
\State{$BestScore := HPipeScore$}
\EndIf
\EndIf
\EndFor
\If{$BestClass\; contains\; Line.class$}
\State \Return 1
\EndIf
\State \Return 0
\EndProcedure
\end{algorithmic}
\caption{HyperPipes Classify Pseudo Code.}\label{pro:Classify}
\end{program}

\begin{program}
\begin{algorithmic}
\Procedure {AddExperience}{$HyperPipes$,$Line$}
\State{$Pipe := FindPipe(Line.class,HyperPipes)$}
\State{$AddThisRow := false$}
\If{$null\; Pipe$}
\State{$HyperPipes := HyperPipes + CreateHyperPipe(Line.class)$}
\State{$Pipe := FindPipe(Line.class,HyperPipes)$}
\EndIf
\ForAll{Attr in Line}
\State{$Pipe.Attr.Num++$}
\State{$Pipe.Attr.Sum += Attr$}
\State{$Pipe.Attr.SumSquares += Attr^2$}
\State{$Pipe.Attr.Mean = Pipe.Attr.Sum/Pipe.Attr.Num$}
\State{$Pipe.Attr.StandardDev = sqrt((Pipe.Attr.SumSquares/Pipe.Attr.Num)-Pipe.Attr.Mean^2)$}
\If{$Pipe.Attr.StandardDev = 0\; ||\; (abs(Pipe.Attr.Mean-Attr)/Pipe.Attr.StandardDev) < 1.96\; ||\; Pipe.Attr.Num < 10$}
\State{$Pipe.Attr.max := max(Pipe.Attr.max,Attr)$}
\State{$Pipe.Attr.min := min(Pipe.Attr.min,Attr)$}
\EndIf
\EndFor
\EndProcedure
\end{algorithmic}
\caption{HyperPipes Add Experience Pseudo Code.}
\label{pro:AddExperience}
\end{program}
